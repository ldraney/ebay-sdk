ebay-sdk User Stories

As a Python developer, I want to call eBay APIs without dealing with auth or raw HTTP

# pip install ldraney-ebay-sdk

from ebay_oauth import EbayOAuthClient
from ebay_sdk import EbayClient

oauth = EbayOAuthClient(
    client_id="my-app-id",
    client_secret="my-secret",
)

with EbayClient(oauth) as ebay:
    # Search for items
    results = ebay.buy_browse.search(q="vintage camera", limit=10)

    # Get category suggestions
    cats = ebay.commerce_taxonomy.get_category_suggestions("0", "camera")

    # List my orders
    orders = ebay.sell_fulfillment.get_orders(limit=5)

As a seller tool developer, I want typed access to all seller APIs

# Manage inventory
ebay.sell_inventory.create_or_replace_inventory_item("MY-SKU-001", {
    "product": {"title": "Vintage Camera"},
    "condition": "USED_EXCELLENT",
    "availability": {
        "shipToLocationAvailability": {"quantity": 3}
    },
})

# Check finances
payouts = ebay.sell_finances.get_payouts(limit=10)
transactions = ebay.sell_finances.get_transactions(limit=10)

# Run marketing campaigns
campaigns = ebay.sell_marketing.get_campaigns(limit=5)

As an MCP server author, I want to wrap this SDK without reimplementing auth or HTTP

# In the MCP server — just wrap SDK methods as tools
from ebay_sdk import EbayClient

@mcp.tool()
def search_ebay(q: str, limit: int = 10):
    """Search eBay listings."""
    return ebay.buy_browse.search(q=q, limit=limit)

@mcp.tool()
def get_order(order_id: str):
    """Get an eBay order by ID."""
    return ebay.sell_fulfillment.get_order(order_id)

# No auth logic, no HTTP logic — the SDK handles it all
